#==============================================================================
# ** Game_System
#------------------------------------------------------------------------------
#  This class handles system data. It saves the disable state of saving and 
# menus. Instances of this class are referenced by $game_system.
#==============================================================================
class Game_System
  #--------------------------------------------------------------------------
  # * Public Instance Variables
  #--------------------------------------------------------------------------
  attr_accessor :window_skin
  #--------------------------------------------------------------------------
  # * Aliasing methods
  #--------------------------------------------------------------------------
  alias doubleD_initialize_change_window initialize
  #--------------------------------------------------------------------------
  # * Object Initialization
  #--------------------------------------------------------------------------
  def initialize(*args)
    doubleD_initialize_change_window(*args)
    @window_skin = "Window"
  end
end
#==============================================================================
# ** Window_Base
#------------------------------------------------------------------------------
#  This is a super class of all windows within the game.
#==============================================================================
class Window_Base
  #--------------------------------------------------------------------------
  # * Aliasing methods
  #--------------------------------------------------------------------------
  alias doubleD_initialize_change_window initialize
  #--------------------------------------------------------------------------
  # * Object Initialization
  #--------------------------------------------------------------------------
  def initialize(*args)
    doubleD_initialize_change_window(*args)
    self.windowskin = Cache.system($game_system.window_skin)
  end
end
#==============================================================================
# ** Game_Interpreter
#------------------------------------------------------------------------------
#  An interpreter for executing event commands. This class is used within the
# Game_Map, Game_Troop, and Game_Event classes.
#==============================================================================
class Game_Interpreter
  #--------------------------------------------------------------------------
  # * function to change the windowksinname
  #--------------------------------------------------------------------------
  def change_windowskin_name(windowskinname)
    p windowskinname
    if Dir.glob("Graphics/System/" + windowskinname)
      $game_system.window_skin = windowskinname
    end
  end
end
#==============================================================================
# ** Scene_Title
#------------------------------------------------------------------------------
#  This class performs the title screen processing.
#==============================================================================
class Scene_Title
  #--------------------------------------------------------------------------
  # * Aliasing methods
  #--------------------------------------------------------------------------
  alias doubleD_start_change_window start
  #--------------------------------------------------------------------------
  # * Start Processing
  #--------------------------------------------------------------------------
  def start(*args)
    $game_system.window_skin = "Window"
    doubleD_start_change_window(*args)
  end
end
